5def explain(ir: dict, language: str = "cobol") -> str: 

6 """ 

7 Generates a natural language explanation strictly based on IR. 

8 This explainer is LIMITED to COBOL and JCL only. 

9  

10 Args: 

11 ir (dict): Intermediate representation containing program structure 

12 language (str): Target language - "cobol" or "jcl" (default: "cobol") 

13  

14 Returns: 

15 str: Natural language explanation of the program 

16  

17 Raises: 

18 ValueError: If language is not COBOL or JCL 

19 TypeError: If ir is not a dictionary 

20 """ 

21 

22 # ------------------------------------------------- 

23 # 🔍 INPUT VALIDATION 

24 # ------------------------------------------------- 

25 if not isinstance(ir, dict): 

26 raise TypeError(f"IR must be a dictionary, got {type(ir).__name__}") 

27 

28 language = language.lower().strip() 

29 

30 if language not in {"cobol", "jcl"}: 

31 raise ValueError( 

32 f"Unsupported language: '{language}'. " 

33 "This explainer supports ONLY COBOL and JCL." 

34 ) 

35 

36 # ------------------------------------------------- 

37 # 🧾 JCL EXPLANATION (NO EXAMPLES – STRUCTURAL ONLY) 

38 # ------------------------------------------------- 

39 if language == "jcl": 

40 return _explain_jcl(ir) 

41 

42 # ------------------------------------------------- 

43 # 🧠 COBOL EXPLANATION WITH SAFE EXAMPLES 

44 # ------------------------------------------------- 

45 return _explain_cobol(ir)

83def _explain_cobol(ir: dict) -> str: 

84 """ 

85 Generates COBOL-specific explanation with conditional section inclusion. 

86  

87 Args: 

88 ir (dict): COBOL intermediate representation 

89  

90 Returns: 

91 str: COBOL explanation 

92 """ 

93 

94 # Extract IR components with safe defaults 

95 program_id = ir.get("program_info", {}).get("program_id", "UNKNOWN") 

96 statements = ir.get("statements", []) 

97 control_flow = ir.get("control_flow", []) 

98 file_ops = ir.get("file_operations", []) 

99 arithmetic_ops = ir.get("arithmetic_operations", []) 

100 

101 # Build dynamic prompt - only include sections with data 

102 prompt = f""" 

103You are a senior IBM Mainframe COBOL engineer. 

104 

105======================== 

106MANDATORY RULES (STRICT) 

107======================== 

108- Explain ONLY what is explicitly present in the IR. 

109- Treat this as STATIC CODE ANALYSIS, not program execution. 

110- Do NOT assume runtime values or evaluate conditions. 

111- Do NOT say which branch "will" or "will not" execute. 

112- Do NOT invent missing statements, paragraphs, or logic. 

113- Do NOT reference line numbers or missing lines. 

114- Use professional COBOL batch terminology only. 

115- Do NOT provide theoretical or hypothetical examples. 

116- Do NOT mention constructs that are not present in the IR. 

117 

118======================== 

119CONTROL FLOW RULES 

120======================== 

121- Explain IF statements in terms of POSSIBLE BRANCHES only if control flow is present. 

122- Use wording such as: 

123 "If the condition is satisfied, the following actions occur..." 

124 "Otherwise, an alternate path is available..." 

125- Never flatten IF/ELSE logic into sequential steps. 

126 

127======================== 

128EXAMPLE RULES 

129======================== 

130- CRITICAL: Give examples ONLY if arithmetic operations appear in the IR section below. 

131- Do NOT provide hypothetical, theoretical, or "what if" examples. 

132- Use ONE simple numeric example per arithmetic operation only. 

133- Do NOT give examples for DISPLAY, MOVE, STOP RUN, PERFORM, or FILE I/O. 

134- If no arithmetic operations section appears below, provide NO examples whatsoever. 

135 

136======================== 

137INPUT (INTERMEDIATE REPRESENTATION) 

138======================== 

139 

140COBOL PROGRAM NAME: 

141{program_id} 

142 

143EXECUTABLE STRUCTURE: 

144{statements} 

145""" 

146 

147 # Conditionally add control flow section 

148 if control_flow: 

149 prompt += f"\nCONTROL FLOW STRUCTURE:\n{control_flow}\n" 

150 

151 # Conditionally add file operations section 

152 if file_ops: 

153 prompt += f"\nFILE OPERATIONS:\n{file_ops}\n" 

154 

155 # Conditionally add arithmetic operations section 

156 if arithmetic_ops: 

157 prompt += f"\nARITHMETIC OPERATIONS:\n{arithmetic_ops}\n" 

158 

159 # Add task and output requirements 

160 prompt += """ 

161======================== 

162TASK 

163======================== 

1641. Describe the overall program purpose at a high level. 

1652. Explain the program flow using semantic blocks (not line-by-line). 

1663. If a CONTROL FLOW STRUCTURE section appears above, explain conditional logic using branch-based descriptions. 

1674. If a FILE OPERATIONS section appears above, describe file handling. 

1685. If an ARITHMETIC OPERATIONS section appears above, mention them and provide ONE example. 

1696. If a section does not appear above, do NOT mention it or provide examples for it. 

170 

171OUTPUT REQUIREMENTS: 

172- One cohesive, professional explanation. 

173- No bullet-point tutorials. 

174- No speculative behavior. 

175- No execution assumptions. 

176- No mention of absent sections. 

177- No hypothetical examples. 

178""" 

179 

180 try: 

181 result = call_llm(prompt).strip() 

182 return result 

183 except Exception as e: 

184 return f"ERROR: Failed to generate COBOL explanation: {str(e)}"
